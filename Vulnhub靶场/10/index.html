1337端口信息提示:

Let’s see how good you are with simple maths Answer my questions 1000 times and I’ll give you your gift

意思是让我们回答1000个问题就给一个大的礼物

用 nc 发起 请求 nc 192.168.245.12 发现是一个计算器,让我们回答1000次数学问题

用python解决即可:

#coding: utf-8

import telnetlib

import time,re



def main():

  try:

​    tn = telnetlib.Telnet('192.168.245.12',port=1337)

  except:

​    print('error')

  time.sleep(0.5)

  loop = 1

  while loop < 1002:

​    data = tn.read_very_eager().decode('ascii')

​    res = re.search('(.*?)\s>',data).group(1)

​    result = str(calc(res))

​    print(str(loop) + ' ' + result)

​    loop = loop + 1

​    tn.write(result.encode('ascii')+b'\n')

​    time.sleep(0.1)

  data = tn.read_very_eager().decode('ascii')

  return data





def calc(res):

  \#(4, '-', 6)

  res_str = res.strip('(').replace(' ','').strip(')')

  nums = res_str.split(',')

  n1 = nums[0].strip()

  operator = nums[1].strip().replace('\'','')

  n2 = nums[2].strip()

  return eval(n1+operator+n2)



print(main())
